X <- c('Happiness', 'Smiled or\nlaughed a lot', 'Enjoyment', 'Worry', 'Sadness', 'Anger')
Y <- c(sum(surveyData$Happiness), sum(surveyData$SmiledALot), sum(surveyData$Enjoyment), sum(surveyData$Worry), sum(surveyData$Sadness), sum(surveyData$Anger))
rep <- as.data.frame(cbind(X, Y))
rep$Y <- as.character(rep$Y)
rep$Y <- as.integer(rep$Y)
rep$pos <- 1:6
pos <- c('Happiness', 'Smiled or\nlaughed a lot', 'Enjoyment')
for(i in 1:6){
  if(rep$X[i] %in% pos)
    rep$pos[i] <- 'Positive Affect'
  else
    rep$pos[i] <- 'Negative Affect'
}
ggplot(data = rep, aes(x = X, y = Y)) + geom_col(aes(fill = rep$pos))+ guides(fill = guide_legend(title = NULL)) + coord_flip() + ylab('count') + xlab('') + scale_y_discrete(limits = seq(0, 130, 10)) + scale_fill_hue(h = c(100, 270), l = c(90, 40)) + ggtitle('Did you experience any of the following during \nA LOT OF THE DAY yesterday?')
